<script>
        const toggleBtn = document.getElementById('toggleEdit');
        const saveBtn = document.getElementById('saveChanges');
        const body = document.body;
        const editableElements = document.querySelectorAll('[contenteditable="true"]');
        
        // --- PENGATURAN KEAMANAN ---
        const PASSWORD_RAHASIA = "leli123"; // Ganti password Anda di sini
        
        // Sembunyikan tombol edit secara default agar orang lain tidak tahu
        toggleBtn.style.display = 'none';

        // CARA MUNCULKAN TOMBOL EDIT: 
        // Klik Logo "Leli Julianti." sebanyak 5 kali dengan cepat
        let clickCount = 0;
        const logo = document.querySelector('.logo');
        logo.addEventListener('click', (e) => {
            e.preventDefault();
            clickCount++;
            if (clickCount === 5) {
                toggleBtn.style.display = 'flex';
                alert("Mode rahasia aktif. Tombol edit muncul di pojok kanan bawah.");
                clickCount = 0;
            }
            setTimeout(() => { clickCount = 0; }, 3000); // Reset hitungan jika tidak lanjut klik dalam 3 detik
        });

        // Fungsi Memuat Konten
        window.onload = () => {
            const savedContent = localStorage.getItem('leli_portfolio_data');
            if (savedContent) {
                const data = JSON.parse(savedContent);
                editableElements.forEach((el, index) => {
                    const id = el.getAttribute('data-id') || `edit-${index}`;
                    if (data[id]) el.innerText = data[id];
                });
            }
            editableElements.forEach(el => el.setAttribute('contenteditable', 'false'));
        };

        // Fungsi Aktifkan Edit dengan Password
        toggleBtn.addEventListener('click', () => {
            const inputPass = prompt("Masukkan Password untuk mengedit:");
            
            if (inputPass === PASSWORD_RAHASIA) {
                body.classList.add('edit-mode');
                editableElements.forEach(el => el.setAttribute('contenteditable', 'true'));
                toggleBtn.style.display = 'none';
                saveBtn.style.display = 'flex';
            } else {
                alert("Password Salah! Akses ditolak.");
            }
        });

        // Fungsi Simpan
        saveBtn.addEventListener('click', () => {
            body.classList.remove('edit-mode');
            editableElements.forEach(el => el.setAttribute('contenteditable', 'false'));
            saveBtn.style.display = 'none';
            toggleBtn.style.display = 'none'; // Sembunyikan lagi setelah simpan
            
            const dataToSave = {};
            editableElements.forEach((el, index) => {
                const id = el.getAttribute('data-id') || `edit-${index}`;
                el.setAttribute('data-id', id);
                dataToSave[id] = el.innerText;
            });
            localStorage.setItem('leli_portfolio_data', JSON.stringify(dataToSave));
            alert("Perubahan berhasil disimpan secara lokal!");
        });
    </script>
